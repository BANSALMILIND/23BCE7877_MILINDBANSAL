-- 1.[JOINS] List the name of each employee along with their job title and department name.
SELECT E.NAME, J.JOB_TITLE, D.DEPT_NAME FROM EMPLOYEES E, JOBS J , DEPARTMENTS D WHERE E.DEPT_ID = D.DEPT_ID AND E.JOB_ID = J.JOB_ID; 

-- 2.[SUBQUERY] Find all employees whose salary is higher than the average salary of the 'IT' department.
SELECT EMP_ID, NAME FROM EMPLOYEES WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES E, DEPARTMENTS D WHERE D.DEPT_ID = E.DEPT_ID AND D.DEPT_NAME = "IT");

-- 3.[GROUP BY] Calculate the total salary expenditure for each department name (not ID).
SELECT D.DEPT_NAME, SUM(E.SALARY) FROM DEPARTMENTS D, EMPLOYEES E WHERE E.DEPT_ID = D.DEPT_ID GROUP BY D.DEPT_NAME;

-- 4.[HAVING] Display departments that have more than 2 employees and an average salary above 70,000.
SELECT D.DEPT_NAME, COUNT(E.EMP_ID) AS TOTAL_EMPLOYEES, AVG(E.SALARY) AS AVERAGE_SALARY FROM DEPARTMENTS D, EMPLOYEES E WHERE D.DEPT_ID = E.DEPT_ID GROUP BY D.DEPT_NAME HAVING COUNT(E.EMP_ID) > 2 AND AVG(E.SALARY) > 70000;

-- 5.[SELF JOIN] Retrieve a list of employee names and their respective manager's names.
SELECT E.NAME AS EMPLOYEE, M.NAME AS MANAGER FROM EMPLOYEES E, EMPLOYEES M WHERE E.MANAGER_ID = M.EMP_ID;

-- 6.[ORDER BY] Find the top 3 highest-paid employees and list them in descending order.
SELECT NAME, SALARY FROM EMPLOYEES ORDER BY SALARY DESC LIMIT 3;

-- 7.[JOINS + WHERE] Find all Developers working in the 'San Francisco' location.
SELECT E.NAME, D.LOCATION, J.JOB_TITLE FROM EMPLOYEES E, JOBS J, DEPARTMENTS D WHERE J.JOB_TITLE = "DEVELOPER" AND D.LOCATION = "SAN FRANCISCO" AND E.DEPT_ID = D.DEPT_ID AND J.JOB_ID = E.JOB_ID;

-- 8.[GROUP BY] Find the number of employees hired each year.
SELECT YEAR(HIRE_DATE) AS HIRE_YEAR, COUNT(EMP_ID) AS EMP_COUNT FROM EMPLOYEES GROUP BY YEAR(HIRE_DATE);

-- 9.[HAVING] List job titles where the maximum salary of an employee in that role is less than 100,000.
SELECT J.JOB_TITLE, MAX(E.SALARY) AS MAX_ROLE_SALARY FROM EMPLOYEES E, JOBS J WHERE E.JOB_ID = J.JOB_ID GROUP BY J.JOB_TITLE HAVING MAX(E.SALARY) < 100000;

-- 10.[SUBQUERY] Find the names of employees who work in the same department as 'David'.
SELECT NAME FROM EMPLOYEES WHERE DEPT_ID = (SELECT DEPT_ID FROM EMPLOYEES WHERE NAME IN ("DAVID"));

-- 11. [MULTI-TABLE JOIN] Show all department names even if they have no employees assigned to them.
SELECT DISTINCT D.DEPT_NAME FROM DEPARTMENTS D LEFT JOIN EMPLOYEES E ON D.DEPT_ID = E.DEPT_ID;

-- 12.[COMPLEX FILTER] Find employees who were hired between 2018 and 2020 and earn more than 60,000.
SELECT NAME FROM EMPLOYEES E WHERE SALARY> 60000 AND YEAR(HIRE_DATE) BETWEEN 2018 AND 2020;

-- 13. [CORRELATED SUBQUERY] Find employees who earn more than the average salary of their own department.

-- 14. [JOINS] Find the job title and salary of the employee with ID 504.
