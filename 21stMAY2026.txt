-- order by, group by 
SELECT * FROM products ORDER BY pid DESC;

SELECT * FROM products ORDER BY pname ASC;

SELECT cid, COUNT(pid) AS total_products FROM products GROUP BY cid;

SELECT cid, MIN(pid) AS lowest_product_id FROM products GROUP BY cid;

SELECT cid, COUNT(pid) AS total_products FROM products GROUP BY cid ORDER BY total_products DESC;

SELECT cid, COUNT(pid) AS total_products FROM products GROUP BY cid HAVING COUNT(pid) >= 1;

SELECT cid, COUNT(pid) AS total_products FROM products GROUP BY cid HAVING COUNT(pid) >= 1 ORDER BY total_products DESC;

SELECT DEPARTMENT, SUM(SALARY) AS accumulated_salary 
FROM WORKER 
GROUP BY DEPARTMENT 
HAVING SUM(SALARY) <= 300000;

SELECT DEPARTMENT, SUM(SALARY) AS total_salary 
FROM WORKER 
GROUP BY DEPARTMENT 
ORDER BY total_salary ASC 
LIMIT 2;

SELECT * FROM WORKER 
WHERE DEPARTMENT = 'Admin' 
ORDER BY SALARY DESC

SELECT w1.DEPARTMENT, MAX(w1.SALARY) AS second_highest_salary
FROM WORKER w1
WHERE w1.SALARY < (
    SELECT MAX(w2.SALARY) 
    FROM WORKER w2 
    WHERE w2.DEPARTMENT = w1.DEPARTMENT
)
GROUP BY w1.DEPARTMENT;
